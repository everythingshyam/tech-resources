  • Script to rename files(Windows)

    This script automates the process of renaming files in a local folder to maintain a common file naming system based on the file creation date and time

    # Define the target directory $directory = "C:\Users\username\Pictures\UnorganzedPictures" # Define the Indian Standard Time zone $istTimeZone = [TimeZoneInfo]::FindSystemTimeZoneById("India Standard Time") # Get all files in the current directory (excluding directories and the script itself) $files = Get-ChildItem -Path $directory -File | Where-Object { $_.Name -ne $MyInvocation.MyCommand.Name } foreach ($file in $files) { # 1. Get the file's creation time and convert it to IST $creationTimeUtc = $file.CreationTimeUtc $creationTimeIst = [TimeZoneInfo]::ConvertTimeFromUtc($creationTimeUtc, $istTimeZone) # 2. Format the timestamp string (e.g., "IMG_20250115_154300") # Note: 3:43 PM is 15:43:00 in 24-hour format (standard for filenames to preserve chronological order) $timestampNoUnderscore = $creationTimeIst.ToString("'IMG'_yyyyMMdd_HHmmss") $fullPrefix = $timestampNoUnderscore + "_" $oldName = $file.Name $newName = "" # Check if the exact timestamp subtext already exists anywhere in the filename if ($oldName -like "*$timestampNoUnderscore*") { # If it's already perfectly at the front, skip it if ($oldName.StartsWith($fullPrefix)) { Write-Host "Skipped (Already prefixed): $oldName" -ForegroundColor Cyan continue } # If it exists elsewhere, strip it out (along with any stray surrounding underscores it might have) # and prepare to move it to the front $cleanedName = $oldName -replace [regex]::Escape($timestampNoUnderscore), "" # Clean up double underscores if the old placement left a gap like "photo__vacation.jpg" $cleanedName = $cleanedName -replace "__", "_" $cleanedName = $cleanedName.TrimStart("_").TrimEnd("_") $newName = $fullPrefix + $cleanedName } else { # If the timestamp isn't in the name at all, just prepend it $newName = $fullPrefix + $oldName } # 3. Rename the file try { Rename-Item -Path $file.FullName -NewName $newName -ErrorAction Stop Write-Host "Renamed: $oldName -> $newName" -ForegroundColor Green } catch { Write-Host "Error renaming $oldName $_" -ForegroundColor Red } }
